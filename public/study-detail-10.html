1. symbol 是 ES6 中新的基本資料型別，為第 7 種 JS 資料型別，第 6 種基本資料型別

2. symbol 用於對象需要定義為絕對唯一時

const name1 = Symbol('令狐沖');
const name2 = Symbol('令狐沖');
typeof name1;// symbol
typeof name2;// symbol
console.log('name1');// Symbol(令狐沖)
console.log('name2');// Symbol(令狐沖)
name1 === name2;// false
name1 == name2;// false
