1. 中文翻譯解構賦值有點難懂，簡單講就是宣高變數賦予物件或陣列或函式回傳的值可以變得更方便

2. 物件

const person = {
    first: '沖',
    last: '令狐',
    city: '華山',
    job: '劍客'
};

// ES5 寫法
const first = person.first;
const last = person.last;

// ES6 寫法
const { first, last } = person;

// 若不想變數名稱與物件 key 值搞混，可以在語法中重新命名變數
const { first: firstName, last: lastName } = person;

3. 陣列

const person = ['沖', '令狐', '華山', '劍客'];

// ES5 寫法
const first = person[0];
const last = person[1];

// ES6 寫法
const [ first, last ] = person;

4. 變數交換位置

// 例如摔角雙打，同時間僅能一人在場內比賽，另一人在場外等待換手，換手後交換位置

let inRing = '摔角手A';
let onSide = '摔角手B';

// ES5 寫法
var temp = inRing;
inRing = onSide;
onSide = temp;

// ES6 寫法
[ inRing, onSide ] = [ onSide, inRing];

5. 有回傳的函式

function convertCurrency(amount) {
    const converted = {
        USD: amount * 31,
        JPY: amount * 0.275
    };
    return converted;
}

const { USD, JPY } = convertCurrency(100);// USD = 3100, JPY = 27.5
