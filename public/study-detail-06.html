1. 陣列常用的 forEach method 一定要跑完陣列的全部，無法中斷 break 或跳過 continue，for of Loop 可解決

const person = [ '令狐沖', '張無忌', '楊過', '韋小寶']

// forEach
person.forEach(cur => {
    if (cur === '張無忌') {
        break;
        console.log(cur);// error
    }
});

// for of
for ( const name of person ) {
    if (name === '張無忌') {
        break;
        console.log(name);// 令狐沖、張無忌
    }
}

2. for of 搭配 Array.entries() 有意想不到的效果

// 承上例
for ( const name of person.entries()) {
    if (name === '張無忌') {
        break;
        console.log(name);// [0, '令狐沖'] [1, '張無忌']
    }
}

// 如果再搭配 Destructuring
for ( const [i, name] of person.entries()) {
    if (name === '張無忌') {
        break;
        console.log(`$ {name}是第 $ {i + 1} 個名字`);// 令狐沖是第 1 個名字，張無忌是第 2 個名字
    }
}
