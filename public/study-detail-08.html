1. spread，與 rest 相反，由少變多

// 用在字串，可將字串的每個字元拆解存成新陣列
const stringSpread = [...'abc 123'];// ['a', 'b', 'c', ' ', '1', '2', '3']

// 用在陣列，可將多個陣列簡寫合併為新陣列，其中並可加入全新元素
const arr1 = ['令狐沖', '張無忌'];
const arr2 = ['任盈盈', '趙敏'];
const newArr1 = [...arr1, ...arr2];// ['令狐沖', '張無忌', '任盈盈', '趙敏']
const newArr2 = [...arr1, '楚留香', ...arr2];// ['令狐沖', '張無忌', '楚留香', '任盈盈', '趙敏']

// 使用...複製陣列後，修改其中一陣列的內容並不會影響另一陣列
// 未使用...複製陣列
const newArr3 = newArr1;
newArr3[0] = '楊過';
newArr3// ['楊過', '張無忌', '任盈盈', '趙敏']
newArr1// ['楊過', '張無忌', '任盈盈', '趙敏']

// 使用...複製陣列
const newArr4 = [...newArr1];
newArr4[0] = '楊過';
newArr4// ['楊過', '張無忌', '任盈盈', '趙敏']
newArr1// ['令狐沖', '張無忌', '任盈盈', '趙敏']

// 用在引數，不想產生新陣列時也可用
arr1.push(...arr2);// ['令狐沖', '張無忌', '任盈盈', '趙敏']

2. rest，與 spread 相反，由多變少

// 用在函式，當傳入的引數數量未知時
function jpyToTwd (rate, ...amounts) {// 日幣轉台幣，rest...結果為陣列
    return amounts.map(amount => amount * rate);// 日幣轉台幣
}
jpyToTwd(0.275, 100, 200, 300);// [27.5, 55, 82.5]
